## promise多次then用法
```js
/**********************多次then 1***********************/
new Promise((resolve, reject) => {
  console.log('p1_内部')
  resolve(1)
}).then(res=>{
  console.log('p_1_then_1', res)
  return 2
}).then(res=>{
  console.log('p1_then_2', res)
})
/**********************多次then 2***********************/
new Promise((resolve, reject) => {
  console.log('p2_内部')
  resolve(1)
}).then(res=>{
  console.log('p_2_then_1', res)
  return new Promise(resolve => {
    resolve(2)
  })
}).then(res=>{
  console.log('p_2_then_2', res)
})
```
